// dueDateService.ts 新增通用方法
export interface DueDateConfig {
  targetDate: Date | string
  referenceDate?: Date
  thresholds?: {
    critical: number  // 临界天数（如7天）
    warning: number   // 警告天数（如30天）
  }
}

export function calculateDueStatus(config: DueDateConfig): DueStatus {
  const target = new Date(config.targetDate)
  const reference = config.referenceDate || new Date()
  const thresholds = config.thresholds || { critical: 7, warning: 30 }

  const daysUntil = Math.ceil((target.getTime() - reference.getTime()) / (1000 * 60 * 60 * 24))

  if (daysUntil <= thresholds.critical) {
    return { status: 'critical', daysUntil, message: '即将到期' }
  } else if (daysUntil <= thresholds.warning) {
    return { status: 'warning', daysUntil, message: '需要注意' }
  } else if (daysUntil < 0) {
    return { status: 'expired', daysUntil, message: '已过期' }
  } else {
    return { status: 'normal', daysUntil, message: '正常' }
  }
}

```typescript
// server/src/utils/crudRoutes.ts
import express, { Router } from 'express'
import { asyncHandler } from '../middleware/errorHandler'
import { logger } from '../middleware/logger'

interface CRUDConfig<T> {
  resourceName: string
  getAll: () => Promise<T[]>
  getById: (id: string) => Promise<T | undefined>
  create: (data: Omit<T, 'id'>) => Promise<T>
  update: (id: string, data: Partial<T>) => Promise<T>
  delete: (id: string) => Promise<void>
}

export function createCRUDRoutes<T>(config: CRUDConfig<T>): Router {
  const router = Router()
  const { resourceName, getAll, getById, create, update, delete: deleteFn } = config

  // GET / - 获取所有资源
  router.get('/', asyncHandler(async (req, res) => {
    logger.info(`Fetching all ${resourceName}s`)
    const items = await getAll()
    const response: ApiResponse<T[]> = {
      success: true,
      data: items,
      timestamp: new Date().toISOString(),
    }
    return res.json(response)
  }))

  // GET /:id - 获取单个资源
  router.get('/:id', asyncHandler(async (req, res) => {
    const { id } = req.params
    logger.info(`Fetching ${resourceName}`, { id })
    const item = await getById(id)

    if (!item) {
      const response: ApiResponse = {
        success: false,
        error: {
          code: `${resourceName.toUpperCase()}_NOT_FOUND`,
          message: `${resourceName}不存在`,
        },
        timestamp: new Date().toISOString(),
      }
      return res.status(404).json(response)
    }

    const response: ApiResponse<T> = {
      success: true,
      data: item,
      timestamp: new Date().toISOString(),
    }
    return res.json(response)
  }))

  // POST / - 创建资源
  router.post('/', asyncHandler(async (req, res) => {
    logger.info(`Creating ${resourceName}`, { body: req.body })
    const item = await create(req.body)
    const response: ApiResponse<T> = {
      success: true,
      data: item,
      timestamp: new Date().toISOString(),
    }
    return res.status(201).json(response)
  }))

  // PUT /:id - 更新资源
  router.put('/:id', asyncHandler(async (req, res) => {
    const { id } = req.params
    logger.info(`Updating ${resourceName}`, { id, body: req.body })
    const item = await update(id, req.body)
    const response: ApiResponse<T> = {
      success: true,
      data: item,
      timestamp: new Date().toISOString(),
    }
    return res.json(response)
  }))

  // DELETE /:id - 删除资源
  router.delete('/:id', asyncHandler(async (req, res) => {
    const { id } = req.params
    logger.info(`Deleting ${resourceName}`, { id })
    await deleteFn(id)
    const response: ApiResponse = {
      success: true,
      data: null,
      timestamp: new Date().toISOString(),
    }
    return res.json(response)
  }))

  return router
}

// 使用示例

  resourceName: 'task',

  create: (data) => taskService.create(data),
  update: (id, data) => taskService.update(id, data),
  delete: (id) => taskService.delete(id),
})
```
