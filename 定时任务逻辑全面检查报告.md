class PreMilestoneWarningJob {
  private timer: NodeJS.Timeout | null = null;

  start() {
    // 1. 立即执行一次（服务启动时）
    this.execute('scheduler');

    // 2. 计算到明天凌晨3:00的延迟
    const now = new Date();
    const tomorrow3AM = new Date(now);
    tomorrow3AM.setDate(tomorrow3AM.getDate() + 1);
    tomorrow3AM.setHours(3, 0, 0, 0);

    const initialDelay = tomorrow3AM.getTime() - now.getTime();

    // 3. 延迟后执行，之后每24小时一次
    setTimeout(() => {
      this.execute('scheduler');
      this.timer = setInterval(() => {
        this.execute('scheduler');
      }, 24 * 60 * 60 * 1000);
    }, initialDelay);
  }
}
```

**检查结果**: ✅ 逻辑正确

- ✅ 之后每24小时执行一次（与cron表达式 `0 3 * * *` 一致）

export async function scanExpiringPermits(): Promise<PermitWarning[]> {
  const { data: permits, error } = await client
    .from('pre_milestones')
    .select('*')
    .not('status', 'in', '("已完成", "已取消")')
    .not('expiry_date', 'is', null)
    .order('expiry_date', { ascending: true })

  for (const permit of permits) {
    const expiryDate = new Date(permit.expiry_date!)
    const daysDiff = Math.ceil((expiryDate.getTime() - today.getTime()) / (1000 * 60 * 60 * 24))

    if (daysDiff <= WARNING_CONFIG.ADVANCE_WARNING_DAYS) {

export async function markAsOverdue(): Promise<number> {
  const today = new Date().toISOString().split('T')[0]

  const { data, error } = await client
    .from('pre_milestones')
    .select('id, milestone_name, status')
    .not('status', 'in', '("已完成", "已取消", "已延期")')
    .lt('expiry_date', today)

  // 更新状态为已延期
  await client.from('pre_milestones').update({
    status: '已延期',
    updated_at: new Date().toISOString()
  }).in('id', data.map(p => p.id))
}
```

#### 建议1: 错峰执行时间优化 ⚠️

**当前执行时间**:

- 自动预警（已禁用）: 02:30

**建议**: 错峰执行时间已合理设置，但建议补充说明文档，说明错峰原因。

**原因**: 避免多个定时任务同时执行导致的数据库负载高峰。

---

#### 建议2: 前期证照预警任务缺少清理逻辑 ⚠️

**当前逻辑**:
```typescript
export async function executeWarningCheck() {
  // 1. 扫描即将过期的证照
  // 2. 创建/更新预警
  // 3. 标记过期的证照
}
```

**建议**: 在执行预警检查后，添加清理已完成证照预警的逻辑。

**建议代码**:
```typescript
export async function executeWarningCheck() {
  // 1. 扫描即将过期的证照
  const expiringPermits = await scanExpiringPermits()

  // 2. 创建/更新预警
  await createWarningsBatch(expiringPermits)

  // 3. 标记过期的证照
  const overdueCount = await markAsOverdue()

  // 4. ✅ 新增：清理已完成证照的预警
  const cleanupCount = await cleanupExpiredWarnings()

  return {
    expiring: expiringPermits.length,
    overdue: overdueCount,
    warningsCreated: expiringPermits.length,
    warningsCleaned: cleanupCount  // 新增
  }
}
```

**原因**: `cleanupExpiredWarnings()` 函数已实现但未被调用。应定期清理已完成/已取消证照的预警记录，避免 `warnings` 表无限增长。
