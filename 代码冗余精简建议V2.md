**精简方案**:
创建通用CRUD路由工厂函数

```typescript
// server/src/utils/createCrudRouter.ts (新建)
import { Router } from 'express'
import { asyncHandler } from '../middleware/errorHandler.js'
import type { ApiResponse } from '../types/index.js'

interface CrudOptions<T> {
  router: Router
  resourceName: string
  serviceName: {
    getAll: (projectId?: string) => Promise<T[]>
    getById: (id: string) => Promise<T | null>
    create: (data: any) => Promise<T>
    update: (id: string, data: any, version?: number) => Promise<T | null>
    delete: (id: string) => Promise<void>
  }
  schemas?: {
    create?: any
    update?: any
  }
}

export function createCrudRouter<T>(options: CrudOptions<T>) {
  const { router, resourceName, serviceName, schemas } = options
  const resourceUpper = resourceName.toUpperCase()

  // GET / - 获取列表
  router.get('/', asyncHandler(async (req, res) => {
    const projectId = req.query.projectId as string | undefined
    const items = await serviceName.getAll(projectId)

    const response: ApiResponse<T[]> = {
      success: true,
      data: items,
      timestamp: new Date().toISOString(),
    }
    res.json(response)
  }))

  // GET /:id - 获取单个
  router.get('/:id', asyncHandler(async (req, res) => {
    const { id } = req.params
    const item = await serviceName.getById(id)

    if (!item) {
      const response: ApiResponse = {
        success: false,
        error: {
          code: `${resourceUpper}_NOT_FOUND`,
          message: `${resourceName}不存在`
        },
        timestamp: new Date().toISOString(),
      }
      return res.status(404).json(response)
    }

    const response: ApiResponse<T> = {
      success: true,
      data: item,
      timestamp: new Date().toISOString(),
    }
    res.json(response)
  }))

  // POST / - 创建
  router.post('/', schemas?.create ? validate(schemas.create) : asyncHandler(async (_, res) => res.next()), asyncHandler(async (req, res) => {
    const item = await serviceName.create({
      ...req.body,
      version: 1,
    })

    const response: ApiResponse<T> = {
      success: true,
      data: item,
      timestamp: new Date().toISOString(),
    }
    res.status(201).json(response)
  }))

  // PUT /:id - 更新
  router.put('/:id', schemas?.update ? validate(schemas.update) : asyncHandler(async (_, res) => res.next()), asyncHandler(async (req, res) => {
    const { id } = req.params
    const { version, ...updates } = req.body

    try {
      const item = await serviceName.update(id, updates, version)

      if (!item) {
        const response: ApiResponse = {
          success: false,
          error: {
            code: `${resourceUpper}_NOT_FOUND`,
            message: `${resourceName}不存在`
          },
          timestamp: new Date().toISOString(),
        }
        return res.status(404).json(response)
      }

      const response: ApiResponse<T> = {
        success: true,
        data: item,
        timestamp: new Date().toISOString(),
      }
      res.json(response)
    } catch (error: any) {
      if (error.message === 'VERSION_MISMATCH') {
        const response: ApiResponse = {
          success: false,
          error: {
            code: 'VERSION_MISMATCH',
            message: '数据已被修改,请刷新后重试'
          },
          timestamp: new Date().toISOString(),
        }
        return res.status(409).json(response)
      }
      throw error
    }
  }))

  // DELETE /:id - 删除
  router.delete('/:id', asyncHandler(async (req, res) => {
    const { id } = req.params
    await serviceName.delete(id)

    const response: ApiResponse = {
      success: true,
      timestamp: new Date().toISOString(),
    }
    res.json(response)
  }))

  return router
}
```

### 4. 错误响应格式重复 - 可减少 ~100 行

**精简方案**:

```typescript
// server/src/utils/responseBuilder.ts (新建)
import type { Response } from 'express'
import type { ApiResponse } from '../types/index.js'

export class ResponseBuilder {
  static success<T>(res: Response, data: T, statusCode = 200): void {
    const response: ApiResponse<T> = {
      success: true,
      data,
      timestamp: new Date().toISOString(),
    }
    res.status(statusCode).json(response)
  }

  static error(
    res: Response,
    statusCode: number,
    code: string,
    message: string
  ): void {
    const response: ApiResponse = {
      success: false,
      error: { code, message },
      timestamp: new Date().toISOString(),
    }
    res.status(statusCode).json(response)
  }

  static notFound(res: Response, resourceName: string): void {
    const resourceUpper = resourceName.toUpperCase()
    this.error(res, 404, `${resourceUpper}_NOT_FOUND`, `${resourceName}不存在`)
  }

  static unauthorized(res: Response, message = '未提供认证token'): void {
    this.error(res, 401, 'UNAUTHORIZED', message)
  }

  static forbidden(res: Response, message = '权限不足'): void {
    this.error(res, 403, 'FORBIDDEN', message)
  }

  static conflict(res: Response, message = '资源冲突'): void {
    this.error(res, 409, 'CONFLICT', message)
  }
}

// 使用示例
router.get('/:id', asyncHandler(async (req, res) => {
  const { id } = req.params
  const task = await supabase.getTask(id)

  if (!task) {
    return ResponseBuilder.notFound(res, 'task')
  }

  ResponseBuilder.success(res, task)
}))
```

```typescript
export function calculateCompletionRatio(completed: number, total: number): number {
  if (total === 0) return 0
  return Math.round((completed / total) * 100)
}

/**
 * 计算风险等级
 */
export function calculateRiskLevel(impact: number, probability: number): 'low' | 'medium' | 'high' {
  const score = impact * probability
  if (score >= 15) return 'high'
  if (score >= 8) return 'medium'
  return 'low'
}

/**
 * 计算健康分数
 */
export function calculateHealthScore(metrics: {
  taskCompletion: number
  riskLevel: string
  milestoneProgress: number
}): number {
  const weights = {
    taskCompletion: 0.4,
    riskLevel: 0.3,
    milestoneProgress: 0.3,
  }

  const riskScores = { low: 1, medium: 0.5, high: 0 }

  const score =
    metrics.taskCompletion * weights.taskCompletion +
    riskScores[metrics.riskLevel as keyof typeof riskScores] * weights.riskLevel * 100 +
    metrics.milestoneProgress * weights.milestoneProgress

  return Math.round(score)
}
```

```typescript
// client/src/hooks/useResourceList.ts (新建)
import { useState, useEffect } from 'react'
import { useToast } from './useToast'

interface UseResourceListOptions<T> {
  fetchFn: () => Promise<T[]>
  deps?: any[]
  enabled?: boolean
}

export function useResourceList<T>(options: UseResourceListOptions<T>) {
  const { fetchFn, deps = [], enabled = true } = options
  const [items, setItems] = useState<T[]>([])
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const toast = useToast()

  const fetch = async () => {
    if (!enabled) return

    setLoading(true)
    setError(null)

    try {
      const data = await fetchFn()
      setItems(data)
    } catch (err: any) {
      setError(err.message || '获取数据失败')
      toast.error(err.message || '获取数据失败')
    } finally {
      setLoading(false)
    }
  }

  useEffect(() => {
    fetch()
  }, [enabled, ...deps])

  const refetch = () => fetch()

  return { items, loading, error, refetch }
}

// 使用示例
function TasksPage() {
  const { items: tasks, loading, error, refetch } = useResourceList<Task>({
    fetchFn: () => api.getTasks(projectId),
    deps: [projectId],
  })

  if (loading) return <Skeleton />
  if (error) return <ErrorMessage error={error} />

  return <TaskList tasks={tasks} onRefresh={refetch} />
}
```

## 📝 注意事项

### 风险点
1. **路由工厂需要支持自定义逻辑**: 某些资源路由可能有特殊需求(如pre-milestones)

1. **逐步重构**: 每次只重构一个模块,确保稳定性

3. **充分测试**: 每个模块重构后立即测试
