2. **健康度计算算法**
   ```typescript
   const calculateHealthScore = () => {
     // 1. 任务完成率权重 40%
     const taskCompletion = (completedTasks / totalTasks) * 0.4;
     
     // 2. 延期任务比例权重 30%
     const delayedTasks = tasks.filter(t => isTaskDelayed(t)).length;
     const delayPenalty = (delayedTasks / totalTasks) * 0.3;
     
     // 3. 高风险数量权重 20%
     const highRiskPenalty = (highRisks / totalRisks) * 0.2;
     
     // 4. 里程碑完成率权重 10%
     const milestoneCompletion = (completedMilestones / totalMilestones) * 0.1;
     
     return Math.min(100, (taskCompletion - delayPenalty - highRiskPenalty + milestoneCompletion) * 100);
   }
   ```

### 阶段3: 自动化预警机制 (P1优先级)
实现房地产系统的自动预警逻辑

#### 实施内容：
1. **定时检测服务**
   ```typescript
   // 每天凌晨自动检测
   const autoCheckRisks = () => {
     // 1. 检测延期任务
     const delayedTasks = tasks.filter(task => {
       const endDate = new Date(task.end_date);
       const today = new Date();
       const diffDays = Math.ceil((endDate.getTime() - today.getTime()) / (1000 * 3600 * 24));
       return diffDays <= 3 && task.progress < 100;
     });
     
     // 2. 检测高风险依赖
     const highDependencyRisks = tasks.filter(task => {
       const dependencyTasks = tasks.filter(t => task.dependencies?.includes(t.id));
       const delayedDependencies = dependencyTasks.filter(d => isTaskDelayed(d));
       return delayedDependencies.length > 0 && task.progress > 0;
     });
     
     // 3. 检测资源冲突
     const resourceConflicts = tasks.filter(task => {
       const sameAssigneeTasks = tasks.filter(t => 
         t.assignee === task.assignee && 
         t.id !== task.id &&
         timeOverlap(task.start_date, task.end_date, t.start_date, t.end_date)
       );
       return sameAssigneeTasks.length > 2;
     });
     
     return {
       delayed: delayedTasks,
       dependency: highDependencyRisks,
       resource: resourceConflicts
     };
   }
   ```

2. **预警分级系统**
   ```typescript
   enum AlertLevel {
     INFO = 'info',      // 信息提醒
     WARNING = 'warning', // 警告
     CRITICAL = 'critical' // 严重
   }
   
   const calculateAlertLevel = (task: Task): AlertLevel => {
     const daysToDeadline = getDaysToDeadline(task);
     
     if (daysToDeadline < 0) {
       return AlertLevel.CRITICAL; // 已延期
     } else if (daysToDeadline <= 1) {
       return AlertLevel.CRITICAL; // 今天到期
     } else if (daysToDeadline <= 3) {
       return AlertLevel.WARNING; // 3天内到期
     } else if (daysToDeadline <= 7) {
       return AlertLevel.INFO; // 1周内到期
     }
     
     return AlertLevel.INFO;
   }
   ```

   class ModalManager {
     private lastShown: Map<string, number> = new Map();
     private cooldownPeriod = 60 * 60 * 1000; // 1小时冷却
     
     canShow(modalId: string): boolean {
       const lastTime = this.lastShown.get(modalId);
       if (!lastTime) return true;
       
       const now = Date.now();
       return now - lastTime > this.cooldownPeriod;
     }
     
     markShown(modalId: string) {
       this.lastShown.set(modalId, Date.now());
     }
   }

     const updatedTask = { ...task };
     
     if (task.progress === 0 && !task.start_date) {
       updatedTask.status = 'not_started';
     } else if (task.progress > 0 && task.progress < 100) {
       updatedTask.status = 'in_progress';
     } else if (task.progress === 100) {
       updatedTask.status = 'completed';
